## file io

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * fs
 * 
 * 
 * 该模块主要用于操作计算机文件
 * 
 ** 文件处理中，大多数都是异步的，因 '处理文件的所需时间' 远远大于 '计算机IO' 的所需时间
 */

const fs = require('fs');


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


/**
 * fs.[method]Sync();
 * 
 * 
 * fs 中大多数方法都会有一个同步执行的 函数( 函数后 添加 Sync )；

 ** 通常，在程序启动时运行有限的次数( 勿在网络请求中使用 )
 */
```

#### fs.reradFile

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * fs.reradFile(path, ?{, options}, callback);
 * 
 * 
 * @?{, options}: 1) 将文件内容以某种编码转换
 *                2) 配置文件转换规则
 * 
 * 根据路径获取 "异步" 指定文件内容
 * return: <Buffer>获取该文件的二进制编码</Buffer>
 * 
 ** 因 '读取文件所需时间' 远远大于 '计算机IO' 的时间，则需要 回调异步处理
 */


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


const fs = require('fs');
const path = require('path');

const filename = path.resolve(__dirname, './demo.txt');

fs.readFile(filename, (err, content) => {

    console.log(content.toString('utf-8'));
});


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


/**
 * readFileSync();
 * 
 * 
 * 根据路径获取 "同步" 指定文件内容
 */

const content = fs.readFileSync(filename, 'utf-8');
console.log(content);

// node 12+ ( promis )
async function test() {

    const content = await fs.promises.readFile(filename, 'utf-8');
    console.log(content);
}

test();


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```
